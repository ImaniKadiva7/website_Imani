// Below I include all the required libraries

#include "algorithm"

#include "vector"

#include "iostream"

using namespace std;

// The function "solution" below which accepts an input as an integer N

// will have the code implemented.

int solution(int N) {

// Using the while statement just below to obtain the binary representation

// of int N. The binary values are then pushed back in array A.

vector int A; // Define array A;

while(N > 0)

{

A.push_back(N % 2); // push back the remainder of N divide by 2

N /= 2; // move to the next value

}

int zero_counter = 0;

int max_length = 0;

// time complexity O(n)

for(int i = A.size() - 1; i >= 0; i--)

{

if(A[i] == 0)

{

zero_counter++;

}

else if(A[i] == 1)

{

max_length = max(max_length, zero_counter);

zero_counter = 0;

}

}

return (max_length);

}
